	OPT triggers swarms, .generate_flag determines normal/alt

engine\events\fish.asm
	determines FISHGROUP_ based on swarm flags

.generate_flag
+;Check if challenge mode is active. If no, don't use alt swarms.
+	ld a, [wChallengeMode]
+	bit GAME_CHALLENGE_MODE_F, a
+	jr z, .normal_swarm
	call Random ; generate a random number below 8
	and %11 ; '3' in bit  (increase bit number with each new added swarm)
	cp 0
	jp z, .normal_swarm
	cp 1
	jp z, .alternate_swarm
	jp .generate_flag
