SetTeamMaxLevel:
	ld a, [wPartyCount]
	ld b, a
	ld hl, wPartyMon1Level
	ld a, [hl]
	dec b
	jr z, .SetLevel
	ld de, PARTYMON_STRUCT_LENGTH
	ld c, a

.LoopPartyLevel
	add hl, de
	ld a, [hl]
	cp c
	jr c, .Continue
	ld c, a
.Continue
	dec b
	jr nz, .LoopPartyLevel
	ld a, c
.SetLevel
	ld b, a
	ld a, [wInBattleTowerBattle]
	and a
	ret nz
	ld a, b
	ld [wTeamMaxLevel], a
	; fallthrough

SetDynamicLevel:
	cp MAX_LEVEL + 1
	ret c
	cp 199
	ret c
	sub LEVEL_FROM_PARTY
	ld b, a
	ld a, [wTeamMaxLevel]
	add b
	cp MAX_LEVEL
	ret c
; cap overflowflow at level 100
	cp LEVEL_FROM_PARTY
	ld a, MAX_LEVEL
	ret c
; cap overflow at level 2
	ld a, 2
	ret
